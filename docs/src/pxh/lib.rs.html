30pub fn get_setting(
31    conn: &Connection,
32    key: &str,
33) -> Result<Option<BString>, Box<dyn std::error::Error>> {
34    let mut stmt = conn.prepare("SELECT value FROM settings WHERE key = ?")?;
35    let mut rows = stmt.query([key])?;
36
37    if let Some(row) = rows.next()? {
38        let value: Vec<u8> = row.get(0)?;
39        Ok(Some(BString::from(value)))
40    } else {
41        Ok(None)
42    }
43}

59pub fn get_hostname() -> BString {
60    let hostname =
61        env::var_os("PXH_HOSTNAME").unwrap_or_else(|| hostname::get().unwrap_or_default());
62
63    // Extract short hostname (before first dot). The shell integration already
64    // sets PXH_HOSTNAME via `hostname -s`, but when falling back to
65    // hostname::get() the result may be an FQDN, so strip the domain suffix.
66    let hostname_bytes = hostname.as_bytes();
67    if let Some(dot_pos) = hostname_bytes.iter().position(|&b| b == b'.') {
68        BString::from(&hostname_bytes[..dot_pos])
69    } else {
70        BString::from(hostname_bytes)
71    }
72}

74pub fn sqlite_connection(path: &Option<PathBuf>) -> Result<Connection, Box<dyn std::error::Error>> {
75    let path = path.as_ref().ok_or("Database not defined; use --db or PXH_DB_PATH")?;
76    let conn = Connection::open(path)?;
77    conn.busy_timeout(Duration::from_millis(5000))?;
78    conn.pragma_update(None, "journal_mode", "WAL")?;
79    conn.pragma_update(None, "temp_store", "MEMORY")?;
80    conn.pragma_update(None, "cache_size", "16777216")?;
81    conn.pragma_update(None, "synchronous", "NORMAL")?;
82
83    let schema = include_str!("base_schema.sql");
84    conn.execute_batch(schema)?;
85
86    // From rusqlite::functions example but adapted for non-utf8
87    // regexps.
88    conn.create_scalar_function("regexp", 2, FunctionFlags::SQLITE_DETERMINISTIC, move |ctx| {
89        assert_eq!(ctx.len(), 2, "called with unexpected number of arguments");
90        let regexp: Arc<Regex> = ctx
91            .get_or_create_aux(0, |vr| -> Result<_, BoxError> { Ok(Regex::new(vr.as_str()?)?) })?;
92        let is_match = {
93            let text = ctx.get_raw(1).as_bytes().map_err(|e| Error::UserFunctionError(e.into()))?;
94
95            regexp.is_match(text)
96        };
97
98        Ok(is_match)
99    })?;
100
101    // Check and set the original_hostname if it's not already set
102    if let Ok(None) = get_setting(&conn, "original_hostname") {
103        let hostname = get_hostname();
104        set_setting(&conn, "original_hostname", &hostname)?;
105    }
106
107    Ok(conn)
108}

160// Try to generate a "stable" session id based on the file imported.
161// If that fails, just create a random one.
162fn generate_import_session_id(histfile: &Path) -> i64 {
163    if let Ok(st) = std::fs::metadata(histfile) {
164        ((st.ino() << 16) | st.dev()) as i64
165    } else {
166        (rand::random::<u64>() >> 1) as i64
167    }

170pub fn import_zsh_history(
171    histfile: &Path,
172    hostname: Option<BString>,
173    username: Option<BString>,
174) -> Result<Vec<Invocation>, Box<dyn std::error::Error>> {
175    let mut f = File::open(histfile)?;
176    let mut buf = Vec::new();
177    let _ = f.read_to_end(&mut buf)?;
178    let username = username
179        .or_else(|| uzers::get_current_username().map(|v| BString::from(v.into_vec())))
180        .unwrap_or_else(|| BString::from("unknown"));
181    let hostname = hostname.unwrap_or_else(get_hostname);
182    let buf_iter = buf.split(|&ch| ch == b'\n');
183
184    let mut ret = vec![];
185    let session_id = generate_import_session_id(histfile);
186    for line in buf_iter {
187        let Some((fields, command)) = line.splitn(2, |&ch| ch == b';').collect_tuple() else {
188            continue;
189        };
190        let Some((_skip, start_time, duration_seconds)) =
191            fields.splitn(3, |&ch| ch == b':').collect_tuple()
192        else {
193            continue;
194        };
195        let start_unix_timestamp = str::from_utf8(&start_time[1..])?.parse::<i64>()?; // 1.. is to skip the leading space!
196        let invocation = Invocation {
197            command: BString::from(command),
198            shellname: "zsh".into(),
199            hostname: Some(BString::from(hostname.as_bytes())),
200            username: Some(BString::from(username.as_bytes())),
201            start_unix_timestamp: Some(start_unix_timestamp),
202            end_unix_timestamp: Some(
203                start_unix_timestamp + str::from_utf8(duration_seconds)?.parse::<i64>()?,
204            ),
205            session_id,
206            ..Default::default()
207        };
208
209        ret.push(invocation);
210    }
211
212    Ok(dedup_invocations(ret))
213}

215pub fn import_bash_history(
216    histfile: &Path,
217    hostname: Option<BString>,
218    username: Option<BString>,
219) -> Result<Vec<Invocation>, Box<dyn std::error::Error>> {
220    let mut f = File::open(histfile)?;
221    let mut buf = Vec::new();
222    let _ = f.read_to_end(&mut buf)?;
223    let username = username
224        .or_else(|| uzers::get_current_username().map(|v| BString::from(v.as_bytes())))
225        .unwrap_or_else(|| BString::from("unknown"));
226    let hostname = hostname.unwrap_or_else(get_hostname);
227    let buf_iter = buf.split(|&ch| ch == b'\n').filter(|l| !l.is_empty());
228
229    let mut ret = vec![];
230    let session_id = generate_import_session_id(histfile);
231    let mut last_ts = None;
232    for line in buf_iter {
233        if line[0] == b'#'
234            && let Ok(ts) = str::parse::<i64>(str::from_utf8(&line[1..]).unwrap_or("0"))
235        {
236            if ts > 0 {
237                last_ts = Some(ts);
238            }
239            continue;
240        }
241        let invocation = Invocation {
242            command: BString::from(line),
243            shellname: "bash".into(),
244            hostname: Some(BString::from(hostname.as_bytes())),
245            username: Some(BString::from(username.as_bytes())),
246            start_unix_timestamp: last_ts,
247            session_id,
248            ..Default::default()
249        };
250
251        ret.push(invocation);
252    }
253
254    Ok(dedup_invocations(ret))
255}

263fn dedup_invocations(invocations: Vec<Invocation>) -> Vec<Invocation> {
264    let mut it = invocations.into_iter();
265    let Some(first) = it.next() else { return vec![] };
266    let mut ret = vec![first];
267    for elem in it {
268        if !elem.sameish(ret.last().unwrap()) {
269            ret.push(elem);
270        }
271    }
272    ret
273}

293// Create a pretty export string that gets serialized as an array of
294// bytes only if it isn't valid UTF-8; this makes the json export
295// prettier.
296#[derive(Debug, Eq, PartialEq, Serialize, Deserialize, Clone)]
297#[serde(untagged)]
298enum PrettyExportString {
299    Readable(String),
300    Encoded(Vec<u8>),
301}
302
303impl From<&[u8]> for PrettyExportString {
304    fn from(bytes: &[u8]) -> Self {
305        match str::from_utf8(bytes) {
306            Ok(v) => Self::Readable(v.to_string()),
307            _ => Self::Encoded(bytes.to_vec()),
308        }
309    }
310}
311
312impl From<Option<&Vec<u8>>> for PrettyExportString {
313    fn from(bytes: Option<&Vec<u8>>) -> Self {
314        match bytes {
315            Some(v) => match str::from_utf8(v.as_slice()) {
316                Ok(s) => Self::Readable(s.to_string()),
317                _ => Self::Encoded(v.to_vec()),
318            },
319            None => Self::Readable(String::new()),
320        }
321    }
322}

355// column list: command, start, host, shell, cwd, end, duratio, session, ...

357struct QueryResultColumnDisplayer {
358    header: &'static str,
359    style: &'static str,
360    displayer: Box<dyn Fn(&Invocation) -> String>,
361}
362
363fn time_display_helper(t: Option<i64>) -> String {
364    // Chained if-let may make this unpacking of
365    // Option/Result/LocalResult cleaner.  Alternative is a closer
366    // using `?` chains but that's slightly uglier.
367    t.and_then(|t| Local.timestamp_opt(t, 0).single())
368        .map(|t| t.format(TIME_FORMAT).to_string())
369        .unwrap_or_else(|| "n/a".to_string())
370}
371
372fn binary_display_helper(v: &BString) -> String {
373    String::from_utf8_lossy(v.as_slice()).to_string()
374}
375
376fn displayers() -> HashMap<&'static str, QueryResultColumnDisplayer> {
377    let mut ret = HashMap::new();
378    ret.insert(
379        "command",
380        QueryResultColumnDisplayer {
381            header: "Command",
382            style: "Fw",
383            displayer: Box::new(|row| binary_display_helper(&row.command)),
384        },
385    );
386    ret.insert(
387        "start_time",
388        QueryResultColumnDisplayer {
389            header: "Start",
390            style: "Fg",
391            displayer: Box::new(|row| time_display_helper(row.start_unix_timestamp)),
392        },
393    );
394    ret.insert(
395        "end_time",
396        QueryResultColumnDisplayer {
397            header: "End",
398            style: "Fg",
399            displayer: Box::new(|row| time_display_helper(row.end_unix_timestamp)),
400        },
401    );
402    ret.insert(
403        "duration",
404        QueryResultColumnDisplayer {
405            header: "Duration",
406            style: "Fm",
407            displayer: Box::new(|row| match (row.start_unix_timestamp, row.end_unix_timestamp) {
408                (Some(start), Some(end)) => format!("{}s", end - start),
409                _ => "n/a".into(),
410            }),
411        },
412    );
413    // TODO: Move the style into the displayer (which would return a
414    // Cell) to allow for color based on per-column values, like red
415    // for non-zero exit statuses.
416    ret.insert(
417        "status",
418        QueryResultColumnDisplayer {
419            header: "Status",
420            style: "Fr",
421            displayer: Box::new(|row| {
422                row.exit_status.map_or_else(|| "n/a".into(), |s| s.to_string())
423            }),
424        },
425    );
426    // TODO: Make session similar to "context" and just print `.` when
427    // it is the current session.
428    ret.insert(
429        "session",
430        QueryResultColumnDisplayer {
431            header: "Session",
432            style: "Fc",
433            displayer: Box::new(|row| format!("{:x}", row.session_id)),
434        },
435    );
436    // Print context specially; the full output is $HOST:$PATH but if
437    // $HOST is the current host, the $HOST: is omitted.  If $PATH is
438    // the current working directory, it is replaced with `.`.
439    ret.insert(
440        "context",
441        QueryResultColumnDisplayer {
442            header: "Context",
443            style: "bFb",
444            displayer: Box::new(|row| {
445                let current_hostname = get_hostname();
446                let row_hostname = row.hostname.clone().unwrap_or_default();
447                let mut ret = String::new();
448                if current_hostname != row_hostname {
449                    write!(ret, "{row_hostname}:").unwrap_or_default();
450                }
451                let current_directory = env::current_dir().unwrap_or_default();
452                ret.push_str(&row.working_directory.as_ref().map_or_else(String::new, |v| {
453                    let v = String::from_utf8_lossy(v.as_slice()).to_string();
454                    if v == current_directory.to_string_lossy() { String::from(".") } else { v }
455                }));
456
457                ret
458            }),
459        },
460    );
461
462    ret
463}

465pub fn present_results_human_readable(
466    fields: &[&str],
467    rows: &[Invocation],
468    suppress_headers: bool,
469) -> Result<(), Box<dyn std::error::Error>> {
470    let displayers = displayers();
471    let mut table = prettytable::Table::new();
472    table.set_format(*prettytable::format::consts::FORMAT_CLEAN);
473
474    if !suppress_headers {
475        let mut title_row = prettytable::Row::empty();
476        for field in fields {
477            let Some(d) = displayers.get(field) else {
478                return Err(Box::from(format!("Invalid 'show' field: {field}")));
479            };
480
481            title_row.add_cell(prettytable::Cell::new(d.header).style_spec("bFg"));
482        }
483        table.set_titles(title_row);
484    }
485
486    for row in rows.iter() {
487        let mut display_row = prettytable::Row::empty();
488        for field in fields {
489            display_row.add_cell(
490                prettytable::Cell::new((displayers[field].displayer)(row).as_str())
491                    .style_spec(displayers[field].style),
492            );
493        }
494        table.add_row(display_row);
495    }
496    table.printstd();
497    Ok(())
498}

500// Rewrite a file with lines matching `contraband` removed.  utf-8
501// safe for the file (TODO: I guess make contraband a `BString` too)
502pub fn atomically_remove_lines_from_file(
503    input_filepath: &PathBuf,
504    contraband: &str,
505) -> Result<(), Box<dyn std::error::Error>> {
506    let input_file = File::open(input_filepath)?;
507    let mut input_reader = BufReader::new(input_file);
508
509    let output_filepath = input_filepath.with_extension(".new"); // good enough for zsh, good enough for us
510    let output_file = File::create(&output_filepath)?;
511    let mut output_writer = BufWriter::new(output_file);
512
513    input_reader.for_byte_line_with_terminator(|line| {
514        if !line.contains_str(contraband) {
515            output_writer.write_all(line)?;
516        }
517        Ok(true)
518    })?;
519
520    output_writer.flush()?;
521    std::fs::rename(output_filepath, input_filepath)?;
522    Ok(())
523}

525/// Rewrite a file removing lines that exactly match any of the contraband items.
526/// Unlike `atomically_remove_lines_from_file`, this matches complete lines (after trimming).
527pub fn atomically_remove_matching_lines_from_file(
528    input_filepath: &Path,
529    contraband_items: &[&str],
530) -> Result<(), Box<dyn std::error::Error>> {
531    use std::collections::HashSet;
532
533    let contraband_set: HashSet<&str> = contraband_items.iter().copied().collect();
534
535    let input_file = File::open(input_filepath)?;
536    let mut input_reader = BufReader::new(input_file);
537
538    let mut output_filepath = input_filepath.as_os_str().to_owned();
539    output_filepath.push(".new");
540    let output_filepath = PathBuf::from(output_filepath);
541    let output_file = File::create(&output_filepath)?;
542    let mut output_writer = BufWriter::new(output_file);
543
544    input_reader.for_byte_line_with_terminator(|line| {
545        let line_str = line.to_str_lossy();
546        let trimmed = line_str.trim();
547        if !contraband_set.contains(trimmed) {
548            output_writer.write_all(line)?;
549        }
550        Ok(true)
551    })?;
552
553    output_writer.flush()?;
554    std::fs::rename(output_filepath, input_filepath)?;
555    Ok(())
556}

559pub mod helpers {
560    use std::path::{Path, PathBuf};

562    /// Parse an SSH command string into command and arguments, handling quotes and spaces.
563    /// Similar to how rsync and other tools parse the -e option.
564    pub fn parse_ssh_command(ssh_cmd: &str) -> (String, Vec<String>) {
565        // If it's a simple command without spaces, just return it
566        if !ssh_cmd.contains(char::is_whitespace) {
567            return (ssh_cmd.to_string(), vec![]);
568        }
569
570        // Otherwise, we need to parse it properly
571        let mut cmd = String::new();
572        let mut args = Vec::new();
573        let mut current = String::new();
574        let mut in_quotes = false;
575        let mut quote_char = '\0';
576        let mut is_first = true;
577        let mut chars = ssh_cmd.chars().peekable();
578
579        while let Some(ch) = chars.next() {
580            match ch {
581                '"' | '\'' if !in_quotes => {
582                    in_quotes = true;
583                    quote_char = ch;
584                }
585                '"' | '\'' if in_quotes && ch == quote_char => {
586                    in_quotes = false;
587                    quote_char = '\0';
588                }
589                ' ' | '\t' if !in_quotes => {
590                    if !current.is_empty() {
591                        if is_first {
592                            cmd = current.clone();
593                            is_first = false;
594                        } else {
595                            args.push(current.clone());
596                        }
597                        current.clear();
598                    }
599                }
600                '\\' if chars.peek().is_some() => {
601                    // Handle escaped characters
602                    if let Some(next_ch) = chars.next() {
603                        current.push(next_ch);
604                    }
605                }
606                _ => {
607                    current.push(ch);
608                }
609            }
610        }
611
612        // Don't forget the last token
613        if !current.is_empty() {
614            if is_first {
615                cmd = current;
616            } else {
617                args.push(current);
618            }
619        }
620
621        (cmd, args)
622    }

624    /// Determines the remote pxh path intelligently. If remote_pxh is
625    /// not "pxh", use it as-is.
626    /// Otherwise, try to determine a smart default based on the current executable location.
627    /// Returns the relative path from home if the binary is in the home directory.
628    pub fn determine_remote_pxh_path(configured_path: &str) -> String {
629        if configured_path != "pxh" {
630            return configured_path.to_string();
631        }
632
633        // Try to be smart about the default path
634        get_relative_path_from_home(None, None).unwrap_or_else(|| "pxh".to_string())
635    }

637    /// Gets the relative path from home directory if the current executable is within it.
638    /// Returns None if the executable is not in the home directory.
639    /// Takes optional overrides for testing.
640    pub fn get_relative_path_from_home(
641        exe_override: Option<&Path>,
642        home_override: Option<&Path>,
643    ) -> Option<String> {
644        let exe = match exe_override {
645            Some(path) => path.to_path_buf(),
646            None => std::env::current_exe().ok()?,
647        };
648
649        let home = match home_override {
650            Some(path) => path.to_path_buf(),
651            None => home::home_dir()?,
652        };
653
654        exe.strip_prefix(&home).ok().map(|path| path.to_string_lossy().to_string())
655    }
656
657    /// Determine if the executable is being invoked as pxhs (shorthand for pxh show)
658    pub fn determine_is_pxhs(args: &[String]) -> bool {
659        args.first()
660            .and_then(|arg| {
661                PathBuf::from(arg).file_name().map(|name| name.to_string_lossy().contains("pxhs"))
662            })
663            .unwrap_or(false)
664    }
665}

679    pub fn pxh_path() -> PathBuf {
680        let mut path = std::env::current_exe().unwrap();
681        path.pop(); // Remove test binary name
682        path.pop(); // Remove 'deps'
683        path.push("pxh");
684        assert!(path.exists(), "pxh binary not found at {:?}", path);
685        path
686    }

688    fn generate_random_string(length: usize) -> String {
689        rand::rng().sample_iter(&Alphanumeric).take(length).map(char::from).collect()
690    }
691
692    fn get_standard_path() -> String {
693        // Use getconf to get the standard PATH
694        Command::new("getconf")
695            .arg("PATH")
696            .output()
697            .ok()
698            .and_then(|output| {
699                if output.status.success() { String::from_utf8(output.stdout).ok() } else { None }
700            })
701            .map(|s| s.trim().to_string())
702            .unwrap_or_else(|| {
703                // Fallback to a reasonable default if getconf fails
704                "/usr/bin:/bin:/usr/sbin:/sbin".to_string()
705            })
706    }

708    /// Unified test helper for invoking pxh with proper isolation and environment setup

717    impl PxhTestHelper {
718        pub fn new() -> Self {
719            let tmpdir = TempDir::new().unwrap();
720            let home_dir = tmpdir.path().to_path_buf();
721            let db_path = home_dir.join(".pxh/pxh.db");

723            PxhTestHelper {

735        }

752        /// Create a pxh command with all environment properly set up
753        pub fn command(&self) -> Command {
754            let mut cmd = Command::new(pxh_path());
755
756            // Clear environment to ensure isolation
757            cmd.env_clear();
758
759            // Set consistent test environment
760            cmd.env("HOME", &self.home_dir);
761            cmd.env("PXH_DB_PATH", &self.db_path);
762            cmd.env("PXH_HOSTNAME", &self.hostname);
763            cmd.env("USER", &self.username);
764            cmd.env("PATH", self.get_full_path());
765
766            // Propagate coverage environment variables if they exist
767            if let Ok(profile_file) = env::var("LLVM_PROFILE_FILE") {
768                cmd.env("LLVM_PROFILE_FILE", profile_file);
769            }
770            if let Ok(llvm_cov) = env::var("CARGO_LLVM_COV") {
771                cmd.env("CARGO_LLVM_COV", llvm_cov);
772            }
773
774            cmd
775        }
776
777        /// Convenience method to create a command with arguments
778        pub fn command_with_args(&self, args: &[&str]) -> Command {
779            let mut cmd = self.command();
780            cmd.args(args);
781            cmd
782        }

784        /// Create a shell command (bash/zsh) with proper environment for interactive testing
785        pub fn shell_command(&self, shell: &str) -> Command {
786            let mut cmd = Command::new(shell);
787
788            // Force interactive mode - use login shell to ensure rc files are loaded
789            cmd.arg("-i");
790            cmd.env_clear();
791
792            // Set consistent environment variables
793            cmd.env("HOME", &self.home_dir);
794            cmd.env("PXH_DB_PATH", &self.db_path);
795            cmd.env("PXH_HOSTNAME", &self.hostname);
796            cmd.env("PATH", self.get_full_path());
797
798            // Set a clean environment for testing
799            cmd.env("USER", &self.username);
800            cmd.env("SHELL", shell);
801
802            // For bash to properly load rc files in a minimal environment
803            cmd.env("BASH_ENV", self.home_dir.join(".bashrc"));
804
805            cmd
806        }
