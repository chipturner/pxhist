27pub fn get_setting(
28    conn: &Connection,
29    key: &str,
30) -> Result<Option<BString>, Box<dyn std::error::Error>> {
31    let mut stmt = conn.prepare("SELECT value FROM settings WHERE key = ?")?;
32    let mut rows = stmt.query([key])?;
33
34    if let Some(row) = rows.next()? {
35        let value: Vec<u8> = row.get(0)?;
36        Ok(Some(BString::from(value)))
37    } else {
38        Ok(None)
39    }
40}

63pub fn sqlite_connection(path: &Option<PathBuf>) -> Result<Connection, Box<dyn std::error::Error>> {
64    let path = path.as_ref().ok_or("Database not defined; use --db or PXH_DB_PATH")?;
65    let conn = Connection::open(path)?;
66    conn.busy_timeout(Duration::from_millis(500))?;
67    conn.pragma_update(None, "journal_mode", "WAL")?;
68    conn.pragma_update(None, "temp_store", "MEMORY")?;
69    conn.pragma_update(None, "cache_size", "16777216")?;
70
71    let schema = include_str!("base_schema.sql");
72    conn.execute_batch(schema)?;
73
74    // From rusqlite::functions example but adapted for non-utf8
75    // regexps.
76    conn.create_scalar_function("regexp", 2, FunctionFlags::SQLITE_DETERMINISTIC, move |ctx| {
77        assert_eq!(ctx.len(), 2, "called with unexpected number of arguments");
78        let regexp: Arc<Regex> = ctx
79            .get_or_create_aux(0, |vr| -> Result<_, BoxError> { Ok(Regex::new(vr.as_str()?)?) })?;
80        let is_match = {
81            let text = ctx.get_raw(1).as_bytes().map_err(|e| Error::UserFunctionError(e.into()))?;
82
83            regexp.is_match(text)
84        };
85
86        Ok(is_match)
87    })?;
88
89    // Check and set the original_hostname if it's not already set
90    if let Ok(None) = get_setting(&conn, "original_hostname") {
91        let hostname = get_hostname();
92        set_setting(&conn, "original_hostname", &hostname)?;
93    }
94
95    Ok(conn)
96}

148// Try to generate a "stable" session id based on the file imported.
149// If that fails, just create a random one.
150fn generate_import_session_id(histfile: &Path) -> i64 {
151    if let Ok(st) = std::fs::metadata(histfile) {
152        ((st.ino() << 16) | st.dev()) as i64
153    } else {
154        (rand::random::<u64>() >> 1) as i64
155    }

158pub fn import_zsh_history(
159    histfile: &Path,
160    hostname: Option<BString>,
161    username: Option<BString>,
162) -> Result<Vec<Invocation>, Box<dyn std::error::Error>> {
163    let mut f = File::open(histfile)?;
164    let mut buf = Vec::new();
165    let _ = f.read_to_end(&mut buf)?;
166    let username = username
167        .or_else(|| users::get_current_username().map(|v| BString::from(v.into_vec())))
168        .unwrap_or_else(|| BString::from("unknown"));
169    let hostname = hostname.unwrap_or_else(get_hostname);
170    let buf_iter = buf.split(|&ch| ch == b'\n');
171
172    let mut ret = vec![];
173    let session_id = generate_import_session_id(histfile);
174    for line in buf_iter {
175        let Some((fields, command)) = line.splitn(2, |&ch| ch == b';').collect_tuple() else {
176            continue;
177        };
178        let Some((_skip, start_time, duration_seconds)) =
179            fields.splitn(3, |&ch| ch == b':').collect_tuple()
180        else {
181            continue;
182        };
183        let start_unix_timestamp = str::from_utf8(&start_time[1..])?.parse::<i64>()?; // 1.. is to skip the leading space!
184        let invocation = Invocation {
185            command: BString::from(command),
186            shellname: "zsh".into(),
187            hostname: Some(BString::from(hostname.as_bytes())),
188            username: Some(BString::from(username.as_bytes())),
189            start_unix_timestamp: Some(start_unix_timestamp),
190            end_unix_timestamp: Some(
191                start_unix_timestamp + str::from_utf8(duration_seconds)?.parse::<i64>()?,
192            ),
193            session_id,
194            ..Default::default()
195        };
196
197        ret.push(invocation);
198    }
199
200    Ok(dedup_invocations(ret))
201}

203pub fn import_bash_history(
204    histfile: &Path,
205    hostname: Option<BString>,
206    username: Option<BString>,
207) -> Result<Vec<Invocation>, Box<dyn std::error::Error>> {
208    let mut f = File::open(histfile)?;
209    let mut buf = Vec::new();
210    let _ = f.read_to_end(&mut buf)?;
211    let username = username
212        .or_else(|| users::get_current_username().map(|v| BString::from(v.as_bytes())))
213        .unwrap_or_else(|| BString::from("unknown"));
214    let hostname = hostname.unwrap_or_else(get_hostname);
215    let buf_iter = buf.split(|&ch| ch == b'\n').filter(|l| !l.is_empty());
216
217    let mut ret = vec![];
218    let session_id = generate_import_session_id(histfile);
219    let mut last_ts = None;
220    for line in buf_iter {
221        if line[0] == b'#' {
222            if let Ok(ts) = str::parse::<i64>(str::from_utf8(&line[1..]).unwrap_or("0")) {
223                if ts > 0 {
224                    last_ts = Some(ts);
225                }
226                continue;
227            }
228        }
229        let invocation = Invocation {
230            command: BString::from(line),
231            shellname: "bash".into(),
232            hostname: Some(BString::from(hostname.as_bytes())),
233            username: Some(BString::from(username.as_bytes())),
234            start_unix_timestamp: last_ts,
235            session_id,
236            ..Default::default()
237        };
238
239        ret.push(invocation);
240    }
241
242    Ok(dedup_invocations(ret))
243}

251fn dedup_invocations(invocations: Vec<Invocation>) -> Vec<Invocation> {
252    let mut it = invocations.into_iter();
253    let Some(first) = it.next() else { return vec![] };
254    let mut ret = vec![first];
255    for elem in it {
256        if !elem.sameish(ret.last().unwrap()) {
257            ret.push(elem);
258        }
259    }
260    ret
261}

281// Create a pretty export string that gets serialized as an array of
282// bytes only if it isn't valid UTF-8; this makes the json export
283// prettier.
284#[derive(Debug, Eq, PartialEq, Serialize, Deserialize, Clone)]
285#[serde(untagged)]
286enum PrettyExportString {
287    Readable(String),
288    Encoded(Vec<u8>),
289}
290
291impl From<&[u8]> for PrettyExportString {
292    fn from(bytes: &[u8]) -> Self {
293        match str::from_utf8(bytes) {
294            Ok(v) => Self::Readable(v.to_string()),
295            _ => Self::Encoded(bytes.to_vec()),
296        }
297    }
298}
299
300impl From<Option<&Vec<u8>>> for PrettyExportString {
301    fn from(bytes: Option<&Vec<u8>>) -> Self {
302        match bytes {
303            Some(v) => match str::from_utf8(v.as_slice()) {
304                Ok(s) => Self::Readable(s.to_string()),
305                _ => Self::Encoded(v.to_vec()),
306            },
307            None => Self::Readable(String::new()),
308        }
309    }
310}

351fn time_display_helper(t: Option<i64>) -> String {
352    // Chained if-let may make this unpacking of
353    // Option/Result/LocalResult cleaner.  Alternative is a closer
354    // using `?` chains but that's slightly uglier.
355    t.and_then(|t| Local.timestamp_opt(t, 0).single())
356        .map(|t| t.format(TIME_FORMAT).to_string())
357        .unwrap_or_else(|| "n/a".to_string())
358}

360fn binary_display_helper(v: &BString) -> String {
361    String::from_utf8_lossy(v.as_slice()).to_string()
362}

364fn displayers() -> HashMap<&'static str, QueryResultColumnDisplayer> {
365    let mut ret = HashMap::new();
366    ret.insert(
367        "command",
368        QueryResultColumnDisplayer {
369            header: "Command",
370            style: "Fw",
371            displayer: Box::new(|row| binary_display_helper(&row.command)),
372        },
373    );
374    ret.insert(
375        "start_time",
376        QueryResultColumnDisplayer {
377            header: "Start",
378            style: "Fg",
379            displayer: Box::new(|row| time_display_helper(row.start_unix_timestamp)),
380        },
381    );
382    ret.insert(
383        "end_time",
384        QueryResultColumnDisplayer {
385            header: "End",
386            style: "Fg",
387            displayer: Box::new(|row| time_display_helper(row.end_unix_timestamp)),
388        },
389    );
390    ret.insert(
391        "duration",
392        QueryResultColumnDisplayer {
393            header: "Duration",
394            style: "Fm",
395            displayer: Box::new(|row| match (row.start_unix_timestamp, row.end_unix_timestamp) {
396                (Some(start), Some(end)) => format!("{}s", end - start),
397                _ => "n/a".into(),
398            }),
399        },
400    );
401    // TODO: Move the style into the displayer (which would return a
402    // Cell) to allow for color based on per-column values, like red
403    // for non-zero exit statuses.
404    ret.insert(
405        "status",
406        QueryResultColumnDisplayer {
407            header: "Status",
408            style: "Fr",
409            displayer: Box::new(|row| {
410                row.exit_status.map_or_else(|| "n/a".into(), |s| s.to_string())
411            }),
412        },
413    );
414    // TODO: Make session similar to "context" and just print `.` when
415    // it is the current session.
416    ret.insert(
417        "session",
418        QueryResultColumnDisplayer {
419            header: "Session",
420            style: "Fc",
421            displayer: Box::new(|row| format!("{:x}", row.session_id)),
422        },
423    );
424    // Print context specially; the full output is $HOST:$PATH but if
425    // $HOST is the current host, the $HOST: is omitted.  If $PATH is
426    // the current working directory, it is replaced with `.`.
427    ret.insert(
428        "context",
429        QueryResultColumnDisplayer {
430            header: "Context",
431            style: "bFb",
432            displayer: Box::new(|row| {
433                let current_hostname = get_hostname();
434                let row_hostname = row.hostname.clone().unwrap_or_default();
435                let mut ret = String::new();
436                if current_hostname != row_hostname {
437                    write!(ret, "{row_hostname}:").unwrap_or_default();
438                }
439                let current_directory = env::current_dir().unwrap_or_default();
440                ret.push_str(&row.working_directory.as_ref().map_or_else(String::new, |v| {
441                    let v = String::from_utf8_lossy(v.as_slice()).to_string();
442                    if v == current_directory.to_string_lossy() { String::from(".") } else { v }
443                }));
444
445                ret
446            }),
447        },
448    );
449
450    ret
451}

453pub fn present_results_human_readable(
454    fields: &[&str],
455    rows: &[Invocation],
456    suppress_headers: bool,
457) -> Result<(), Box<dyn std::error::Error>> {
458    let displayers = displayers();
459    let mut table = prettytable::Table::new();
460    table.set_format(*prettytable::format::consts::FORMAT_CLEAN);
461
462    if !suppress_headers {
463        let mut title_row = prettytable::Row::empty();
464        for field in fields {
465            let Some(d) = displayers.get(field) else {
466                return Err(Box::from(format!("Invalid 'show' field: {field}")));
467            };
468
469            title_row.add_cell(prettytable::Cell::new(d.header).style_spec("bFg"));
470        }
471        table.set_titles(title_row);
472    }
473
474    for row in rows.iter() {
475        let mut display_row = prettytable::Row::empty();
476        for field in fields {
477            display_row.add_cell(
478                prettytable::Cell::new((displayers[field].displayer)(row).as_str())
479                    .style_spec(displayers[field].style),
480            );
481        }
482        table.add_row(display_row);
483    }
484    table.printstd();
485    Ok(())
486}

488// Rewrite a file with lines matching `contraband` removed.  utf-8
489// safe for the file (TODO: I guess make contraband a `BString` too)
490pub fn atomically_remove_lines_from_file(
491    input_filepath: &PathBuf,
492    contraband: &str,
493) -> Result<(), Box<dyn std::error::Error>> {
494    let input_file = File::open(input_filepath)?;
495    let mut input_reader = BufReader::new(input_file);
496
497    let output_filepath = input_filepath.with_extension(".new"); // good enough for zsh, good enough for us
498    let output_file = File::create(&output_filepath)?;
499    let mut output_writer = BufWriter::new(output_file);
500
501    input_reader.for_byte_line_with_terminator(|line| {
502        if !line.contains_str(contraband) {
503            output_writer.write_all(line)?;
504        }
505        Ok(true)
506    })?;
507
508    output_writer.flush()?;
509    std::fs::rename(output_filepath, input_filepath)?;
510    Ok(())
511}

517    /// Parse an SSH command string into command and arguments, handling quotes and spaces.
518    /// Similar to how rsync and other tools parse the -e option.
519    pub fn parse_ssh_command(ssh_cmd: &str) -> (String, Vec<String>) {
520        // If it's a simple command without spaces, just return it
521        if !ssh_cmd.contains(char::is_whitespace) {
522            return (ssh_cmd.to_string(), vec![]);
523        }
524
525        // Otherwise, we need to parse it properly
526        let mut cmd = String::new();
527        let mut args = Vec::new();
528        let mut current = String::new();
529        let mut in_quotes = false;
530        let mut quote_char = '\0';
531        let mut is_first = true;
532        let mut chars = ssh_cmd.chars().peekable();
533
534        while let Some(ch) = chars.next() {
535            match ch {
536                '"' | '\'' if !in_quotes => {
537                    in_quotes = true;
538                    quote_char = ch;
539                }
540                '"' | '\'' if in_quotes && ch == quote_char => {
541                    in_quotes = false;
542                    quote_char = '\0';
543                }
544                ' ' | '\t' if !in_quotes => {
545                    if !current.is_empty() {
546                        if is_first {
547                            cmd = current.clone();
548                            is_first = false;
549                        } else {
550                            args.push(current.clone());
551                        }
552                        current.clear();
553                    }
554                }
555                '\\' if chars.peek().is_some() => {
556                    // Handle escaped characters
557                    if let Some(next_ch) = chars.next() {
558                        current.push(next_ch);
559                    }
560                }
561                _ => {
562                    current.push(ch);
563                }
564            }
565        }
566
567        // Don't forget the last token
568        if !current.is_empty() {
569            if is_first {
570                cmd = current;
571            } else {
572                args.push(current);
573            }
574        }
575
576        (cmd, args)
577    }

579    /// Determines the remote pxh path intelligently. If remote_pxh is
580    /// not "pxh", use it as-is.
581    /// Otherwise, try to determine a smart default based on the current executable location.
582    /// Returns the relative path from home if the binary is in the home directory.
583    pub fn determine_remote_pxh_path(configured_path: &str) -> String {
584        if configured_path != "pxh" {
585            return configured_path.to_string();
586        }
587
588        // Try to be smart about the default path
589        get_relative_path_from_home(None, None).unwrap_or_else(|| "pxh".to_string())

592    /// Gets the relative path from home directory if the current executable is within it.
593    /// Returns None if the executable is not in the home directory.
594    /// Takes optional overrides for testing.
595    pub fn get_relative_path_from_home(
596        exe_override: Option<&Path>,
597        home_override: Option<&Path>,
598    ) -> Option<String> {
599        let exe = match exe_override {
600            Some(path) => path.to_path_buf(),
601            None => std::env::current_exe().ok()?,
602        };
603
604        let home = match home_override {
605            Some(path) => path.to_path_buf(),
606            None => home::home_dir()?,
607        };
608
609        exe.strip_prefix(&home).ok().map(|path| path.to_string_lossy().to_string())
610    }

613    pub fn determine_is_pxhs(args: &[String]) -> bool {
614        args.first()
615            .and_then(|arg| {
616                PathBuf::from(arg).file_name().map(|name| name.to_string_lossy().contains("pxhs"))
617            })
618            .unwrap_or(false)

648        // Use getconf to get the standard PATH
649        Command::new("getconf")

657            .unwrap_or_else(|| {
658                // Fallback to a reasonable default if getconf fails
659                "/usr/bin:/bin:/usr/sbin:/sbin".to_string()
660            })
661    }

663    /// Unified test helper for invoking pxh with proper isolation and environment setup
664    pub struct PxhTestHelper {
665        _tmpdir: TempDir,
666        pub hostname: String,
667        pub username: String,
668        home_dir: PathBuf,
669        db_path: PathBuf,
670    }
671
672    impl PxhTestHelper {
673        pub fn new() -> Self {
674            let tmpdir = TempDir::new().unwrap();
675            let home_dir = tmpdir.path().to_path_buf();
676            let db_path = home_dir.join(".pxh/pxh.db");
677
678            PxhTestHelper {
679                _tmpdir: tmpdir,
680                hostname: generate_random_string(12),
681                username: "testuser".to_string(),
682                home_dir,
683                db_path,
684            }
685        }
686
687        pub fn with_custom_db_path(mut self, db_path: impl AsRef<Path>) -> Self {
688            self.db_path = self.home_dir.join(db_path);
689            self
690        }
691
692        /// Get the temporary directory path
693        pub fn home_dir(&self) -> &Path {
694            &self.home_dir
695        }
696
697        /// Get the database path
698        pub fn db_path(&self) -> &Path {
699            &self.db_path
700        }
701
702        /// Get the full PATH including pxh binary directory
703        pub fn get_full_path(&self) -> String {
704            format!("{}:{}", pxh_path().parent().unwrap().display(), get_standard_path())
705        }
706
707        /// Create a pxh command with all environment properly set up
708        pub fn command(&self) -> Command {
709            let mut cmd = Command::new(pxh_path());
710
711            // Clear environment to ensure isolation
712            cmd.env_clear();
713
714            // Set consistent test environment
715            cmd.env("HOME", &self.home_dir);
716            cmd.env("PXH_DB_PATH", &self.db_path);
717            cmd.env("PXH_HOSTNAME", &self.hostname);
718            cmd.env("USER", &self.username);
719            cmd.env("PATH", self.get_full_path());
720
721            // Propagate coverage environment variables if they exist
722            if let Ok(profile_file) = env::var("LLVM_PROFILE_FILE") {
723                cmd.env("LLVM_PROFILE_FILE", profile_file);
724            }
725            if let Ok(llvm_cov) = env::var("CARGO_LLVM_COV") {
726                cmd.env("CARGO_LLVM_COV", llvm_cov);
727            }
728
729            cmd
730        }
731
732        /// Convenience method to create a command with arguments
733        pub fn command_with_args(&self, args: &[&str]) -> Command {
734            let mut cmd = self.command();
735            cmd.args(args);
736            cmd
737        }
738
739        /// Create a shell command (bash/zsh) with proper environment for interactive testing
740        pub fn shell_command(&self, shell: &str) -> Command {
741            let mut cmd = Command::new(shell);
742
743            // Force interactive mode - use login shell to ensure rc files are loaded
744            cmd.arg("-i");
745            cmd.env_clear();
746
747            // Set consistent environment variables
748            cmd.env("HOME", &self.home_dir);
749            cmd.env("PXH_DB_PATH", &self.db_path);
750            cmd.env("PXH_HOSTNAME", &self.hostname);
751            cmd.env("PATH", self.get_full_path());
752
753            // Set a clean environment for testing
754            cmd.env("USER", &self.username);
755            cmd.env("SHELL", shell);
756
757            // For bash to properly load rc files in a minimal environment
758            cmd.env("BASH_ENV", self.home_dir.join(".bashrc"));
759
760            cmd
761        }
762    }
